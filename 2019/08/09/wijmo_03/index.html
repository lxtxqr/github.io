Wijmo-FlexGrid的全选择和全解除的实现

技术 Wijmo 2019/08/09

首先，在WijimoJS的组件中，FlexGrid是最常用的数据集合组件。
FlexGrid中如何绑定一个Checkbox呢，其实很简单
加入第一列是一个Checkbox，绑定的数字元素叫No。
只需要把这个No元素定义为Boolean类型，就可以自动显示出了CheckBox。

以下的例子代码是FlexGrid的全选择和接触的解决方法
HTML代码

<wj-flex-grid [headersVisibility]="'Column'" [itemsSource]="data"
  class="max-ht-315" [allowSorting]="false" [allowDragging]="'None'" #flexGrid>
  <wj-flex-grid-column [header]="'No.'" [binding]="'no'" [width]="130" [dataType]="2" isReadOnly="true">
</wj-flex-grid-column>
<wj-flex-grid-column [header]="'AAAA'" [binding]="'no'" [width]="130" [dataType]="2" isReadOnly="true">
    <ng-template wjFlexGridCellTemplate [cellType]="'Cell'" let-item="item" let-cell="cell">
        <div>BBBB</div>
    </ng-template>
    <ng-template wjFlexGridCellTemplate [cellType]="'CellEdit'">
    </ng-template>
 </wj-flex-grid-column>
</wj-flex-grid>

<button #select type="button" class="gyoumu-button" (click)="zenSelect()">全选择</button>
<button type="button" class="gyoumu-button" (click)="keijyo()">全解除</button>

TS代码


import { Component, OnInit, ViewChild, AfterViewInit} from '@angular/core';
import { WjFlexGrid } from 'wijmo/wijmo.angular2.grid';

@Component({
  selector: 'app-xxx0201',
  templateUrl: './xxx0201.component.html',
  styleUrls: ['./xxx0201.component.css']
})

export class xxx0201Component implements OnInit {
  @ViewChild('flexGrid') flexGrid: WjFlexGrid;

  data = [
    {
      number: '1',
      select: true
    },
    {
      number: '2',
      select: false
    },
    {
      number: '3',
      select: false
    },
    {
      number: '4',
      select: false
    }
  ];

  constructor() {

  }

  ngOnInit() {
  }
  // 全选择
  zenSelect() {
    const collectionView = this.flexGrid.collectionView;
    const  items = collectionView.items;

    for (let i = 0; i < items.length; i++) {
      items[i].select = true;
    }
    this.flexGrid.refresh();

  }
  // 全解除
  keijyo() {
    const collectionView = this.flexGrid.collectionView;
    const  items = collectionView.items;

    for (let i = 0; i < items.length; i++) {
      items[i].select = false;
    }
    this.flexGrid.refresh();
  }
}
